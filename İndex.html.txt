    <script>
        gsap.registerPlugin(ScrollTrigger);
        
        const bgImages = document.querySelectorAll('.hero-bg-image');
        let currentImg = 0;
        
        function rotateBackground() {
            bgImages[currentImg].classList.remove('active');
            currentImg = (currentImg + 1) % bgImages.length;
            bgImages[currentImg].classList.add('active');
        }
        
        if(bgImages.length > 0) {
            setInterval(rotateBackground, 6000);
        }

        window.addEventListener('load', () => {
            const tl = gsap.timeline();
            tl.to(".loader-text", { y: 0, opacity: 1, duration: 1 })
              .to(".loader-sub", { y: 0, opacity: 1, duration: 0.8 }, "-=0.5")
              .to("#loader", { yPercent: -100, duration: 1.2, ease: "expo.inOut", delay: 0.5 })
              .to("#main-content", { opacity: 1, duration: 0.5 }, "-=0.5")
              .to(".hero-line h1, .hero-line h2, .hero-line p", { y: 0, duration: 1, stagger: 0.1, ease: "power4.out" })
              .to(".hero-btn", { opacity: 1, y: 0, duration: 0.8 }, "-=0.5");
            
            document.body.classList.remove('loading');
        });

        const menuBtn = document.getElementById('menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');
        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('open');
            document.body.style.overflow = mobileMenu.classList.contains('open') ? 'hidden' : 'auto';
        });

        gsap.utils.toArray('.reveal-trigger').forEach(el => {
            gsap.from(el.children, {
                y: 50, opacity: 0, duration: 1, stagger: 0.2, scrollTrigger: { trigger: el, start: "top 85%" }
            });
        });

        gsap.from('.card-reveal', {
            y: 50, opacity: 0, duration: 1, stagger: 0.2, scrollTrigger: { trigger: '#classes', start: "top 75%" }
        });
